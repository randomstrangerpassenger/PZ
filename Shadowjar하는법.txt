$root = "C:\Users\MW\Downloads\coding\PZ"
$output = Join-Path $root "Pulse_All_Source.txt"

# UTF-8 (no BOM) 고정 StreamWriter
$utf8NoBom = New-Object System.Text.UTF8Encoding($false)
$sw = New-Object System.IO.StreamWriter($output, $false, $utf8NoBom)

try {
  Get-ChildItem $root -Recurse -File |
    Where-Object {
      $_.Extension -match '\.(java|kt|lua|json|xml|properties|gradle)$' -and
      $_.FullName -notmatch '\\(build|out|\.gradle|\.idea|libs|run|logs|bin)\\'
    } |
    Sort-Object FullName |
    ForEach-Object {
      $sw.WriteLine("")
      $sw.WriteLine("")
      $sw.WriteLine("===== FILE: $($_.FullName) =====")
      $sw.WriteLine("")

      # 파일을 '그대로' 읽기
      $text = [System.IO.File]::ReadAllText($_.FullName, [System.Text.Encoding]::UTF8)
      $sw.Write($text)
    }
}
finally {
  $sw.Flush()
  $sw.Close()
}
