$root = "C:\Users\MW\Downloads\coding\PZ"
$output = "$root\Pulse_All_Source.txt"

Remove-Item $output -ErrorAction SilentlyContinue

Get-ChildItem $root -Recurse -File |
Where-Object {
    $_.Extension -match '\.(java|kt|lua|json|xml|properties|gradle)$' -and
    $_.FullName -notmatch '\\(build|out|\.gradle|\.idea|libs|run|logs|bin)\\'
} |
Sort-Object FullName |
ForEach-Object {
    Add-Content $output "`n`n===== FILE: $($_.FullName) =====`n"
    Get-Content $_.FullName -Raw | Add-Content $output
}
