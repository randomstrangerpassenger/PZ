Set-Location "C:\Users\MW\Downloads\coding\Echo"

# 기존 파일이 있으면 삭제하여 초기화
Remove-Item "Echo_All_Code.txt" -ErrorAction Ignore

Get-ChildItem -Recurse -Include *.java, *.kt, *.json, *.xml, *.gradle |
Sort-Object FullName |
ForEach-Object {
    Add-Content -Path "Echo_All_Code.txt" -Value "`n===== FILE START: $($_.FullName) =====`n"
    Get-Content $_.FullName | Add-Content -Path "Echo_All_Code.txt"
    Add-Content -Path "Echo_All_Code.txt" -Value "`n===== FILE END: $($_.FullName) =====`n"
}
